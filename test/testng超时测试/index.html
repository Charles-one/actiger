TestNG超时测试

超出预期时间则用例失败

package com.actiger.test.timeout;

import org.testng.annotations.Test;

public class TimeOut {

    @Test(timeOut = 3000)
    public void testSuccess() throws InterruptedException{
        Thread.sleep(2000);
    }


    @Test(timeOut = 2000)
    public void testFail() throws InterruptedException{
        Thread.sleep(3000);
    }

}
